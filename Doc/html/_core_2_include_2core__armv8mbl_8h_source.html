1703
1704 return (
1705 ((PreemptPriority & (uint32_t)((1UL << (PreemptPriorityBits)) - 1UL)) << SubPriorityBits) |
1706 ((SubPriority & (uint32_t)((1UL << (SubPriorityBits )) - 1UL)))
1707 );
