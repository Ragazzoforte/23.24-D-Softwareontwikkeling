4104
4105 return (
4106 ((PreemptPriority & (uint32_t)((1UL << (PreemptPriorityBits)) - 1UL)) << SubPriorityBits) |
4107 ((SubPriority & (uint32_t)((1UL << (SubPriorityBits )) - 1UL)))
4108 );
