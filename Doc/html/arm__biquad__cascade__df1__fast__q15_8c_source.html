60void arm_biquad_cascade_df1_fast_q15(
61 const arm_biquad_casd_df1_inst_q15 * S,
62 const q15_t * pSrc,
63 q15_t * pDst,
64 uint32_t blockSize)
65{
66 const q15_t *pIn = pSrc; /* Source pointer */
67 q15_t *pOut = pDst; /* Destination pointer */
68 q15_t *pState = S->pState; /* State pointer */
69 const q15_t *pCoeffs = S->pCoeffs; /* Coefficient pointer */
70 q31_t acc; /* Accumulator */
71 q31_t in; /* Temporary variable to hold input value */
72 q31_t out; /* Temporary variable to hold output value */
73 q31_t b0; /* Temporary variable to hold bo value */
74 q31_t b1, a1; /* Filter coefficients */
75 q31_t state_in, state_out; /* Filter state variables */
76 int32_t shift = (int32_t) (15 - S->postShift); /* Post shift */
77 uint32_t sample, stage = S->numStages; /* Loop counters */
78
79 do
80 {
81 /* Read the b0 and 0 coefficients using SIMD */
82 b0 = read_q15x2_ia (&pCoeffs);
83
84 /* Read the b1 and b2 coefficients using SIMD */
85 b1 = read_q15x2_ia (&pCoeffs);
86
87 /* Read the a1 and a2 coefficients using SIMD */
88 a1 = read_q15x2_ia (&pCoeffs);
89
90 /* Read the input state values from the state buffer: x[n-1], x[n-2] */
91 state_in = read_q15x2_ia (&pState);
92
93 /* Read the output state values from the state buffer: y[n-1], y[n-2] */
94 state_out = read_q15x2_da (&pState);
95
96#if defined (ARM_MATH_LOOPUNROLL)
97
98 /* Apply loop unrolling and compute 2 output values simultaneously. */
99 /* Variable acc hold output values that are being computed:
100 *
101 * acc = b0 * x[n] + b1 * x[n-1] + b2 * x[n-2] + a1 * y[n-1] + a2 * y[n-2]
102 * acc = b0 * x[n] + b1 * x[n-1] + b2 * x[n-2] + a1 * y[n-1] + a2 * y[n-2]
103 */
104
105 /* Loop unrolling: Compute 2 outputs at a time */
106 sample = blockSize >> 1U;
107
108 while (sample > 0U)
109 {
110
111 /* Read the input */
112 in = read_q15x2_ia (&pIn);
113
114 /* out = b0 * x[n] + 0 * 0 */
115 out = __SMUAD(b0, in);
116 /* acc = b1 * x[n-1] + acc += b2 * x[n-2] + out */
117 acc = __SMLAD(b1, state_in, out);
118 /* acc += a1 * y[n-1] + acc += a2 * y[n-2] */
119 acc = __SMLAD(a1, state_out, acc);
120
121 /* The result is converted from 3.29 to 1.31 and then saturation is applied */
122 out = __SSAT((acc >> shift), 16);
123
124 /* Every time after the output is computed state should be updated. */
125 /* The states should be updated as: */
126 /* Xn2 = Xn1 */
127 /* Xn1 = Xn */
128 /* Yn2 = Yn1 */
129 /* Yn1 = acc */
130 /* x[n-N], x[n-N-1] are packed together to make state_in of type q31 */
131 /* y[n-N], y[n-N-1] are packed together to make state_out of type q31 */
132
133#ifndef ARM_MATH_BIG_ENDIAN
134 state_in = __PKHBT(in, state_in, 16);
135 state_out = __PKHBT(out, state_out, 16);
136#else
137 state_in = __PKHBT(state_in >> 16, (in >> 16), 16);
138 state_out = __PKHBT(state_out >> 16, (out), 16);
139#endif /* #ifndef ARM_MATH_BIG_ENDIAN */
140
141 /* out = b0 * x[n] + 0 * 0 */
142 out = __SMUADX(b0, in);
143 /* acc0 = b1 * x[n-1] , acc0 += b2 * x[n-2] + out */
144 acc = __SMLAD(b1, state_in, out);
145 /* acc += a1 * y[n-1] + acc += a2 * y[n-2] */
146 acc = __SMLAD(a1, state_out, acc);
147
148 /* The result is converted from 3.29 to 1.31 and then saturation is applied */
149 out = __SSAT((acc >> shift), 16);
150
151 /* Store the output in the destination buffer. */
152#ifndef ARM_MATH_BIG_ENDIAN
153 write_q15x2_ia (&pOut, __PKHBT(state_out, out, 16));
154#else
155 write_q15x2_ia (&pOut, __PKHBT(out, state_out >> 16, 16));
156#endif /* #ifndef ARM_MATH_BIG_ENDIAN */
157
158 /* Every time after the output is computed state should be updated. */
159 /* The states should be updated as: */
160 /* Xn2 = Xn1 */
161 /* Xn1 = Xn */
162 /* Yn2 = Yn1 */
163 /* Yn1 = acc */
164 /* x[n-N], x[n-N-1] are packed together to make state_in of type q31 */
165 /* y[n-N], y[n-N-1] are packed together to make state_out of type q31 */
166#ifndef ARM_MATH_BIG_ENDIAN
167 state_in = __PKHBT(in >> 16, state_in, 16);
168 state_out = __PKHBT(out, state_out, 16);
169#else
170 state_in = __PKHBT(state_in >> 16, in, 16);
171 state_out = __PKHBT(state_out >> 16, out, 16);
172#endif /* #ifndef ARM_MATH_BIG_ENDIAN */
173
174 /* Decrement loop counter */
175 sample--;
176 }
177
178 /* Loop unrolling: Compute remaining outputs */
179 sample = (blockSize & 0x1U);
180
181#else
182
183 /* Initialize blkCnt with number of samples */
184 sample = blockSize;
185
186#endif /* #if defined (ARM_MATH_LOOPUNROLL) */
187
188 while (sample > 0U)
189 {
190 /* Read the input */
191 in = *pIn++;
192
193 /* out = b0 * x[n] + 0 * 0 */
194#ifndef ARM_MATH_BIG_ENDIAN
195 out = __SMUAD(b0, in);
196#else
197 out = __SMUADX(b0, in);
198#endif /* #ifndef ARM_MATH_BIG_ENDIAN */
199
200 /* acc = b1 * x[n-1], acc += b2 * x[n-2] + out */
201 acc = __SMLAD(b1, state_in, out);
202 /* acc += a1 * y[n-1] + acc += a2 * y[n-2] */
203 acc = __SMLAD(a1, state_out, acc);
204
205 /* The result is converted from 3.29 to 1.31 and then saturation is applied */
206 out = __SSAT((acc >> shift), 16);
207
208 /* Store the output in the destination buffer. */
209 *pOut++ = (q15_t) out;
210
211 /* Every time after the output is computed state should be updated. */
212 /* The states should be updated as: */
213 /* Xn2 = Xn1 */
214 /* Xn1 = Xn */
215 /* Yn2 = Yn1 */
216 /* Yn1 = acc */
217 /* x[n-N], x[n-N-1] are packed together to make state_in of type q31 */
218 /* y[n-N], y[n-N-1] are packed together to make state_out of type q31 */
219#ifndef ARM_MATH_BIG_ENDIAN
220 state_in = __PKHBT(in, state_in, 16);
221 state_out = __PKHBT(out, state_out, 16);
222#else
223 state_in = __PKHBT(state_in >> 16, in, 16);
224 state_out = __PKHBT(state_out >> 16, out, 16);
225#endif /* #ifndef ARM_MATH_BIG_ENDIAN */
226
227 /* decrement loop counter */
228 sample--;
229 }
230
231 /* The first stage goes from the input buffer to the output buffer. */
232 /* Subsequent (numStages - 1) occur in-place in the output buffer */
233 pIn = pDst;
234
235 /* Reset the output pointer */
236 pOut = pDst;
237
238 /* Store the updated state variables back into the state array */
239 write_q15x2_ia(&pState, state_in);
240 write_q15x2_ia(&pState, state_out);
241
242 /* Decrement loop counter */
243 stage--;
244
245 } while (stage > 0U);
246}
