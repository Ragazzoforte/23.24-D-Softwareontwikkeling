49void arm_fir_f64(
50 const arm_fir_instance_f64 * S,
51 const float64_t * pSrc,
52 float64_t * pDst,
53 uint32_t blockSize)
54{
55 float64_t *pState = S->pState; /* State pointer */
56 const float64_t *pCoeffs = S->pCoeffs; /* Coefficient pointer */
57 float64_t *pStateCurnt; /* Points to the current sample of the state */
58 float64_t *px; /* Temporary pointer for state buffer */
59 const float64_t *pb; /* Temporary pointer for coefficient buffer */
60 float64_t acc0; /* Accumulator */
61 uint32_t numTaps = S->numTaps; /* Number of filter coefficients in the filter */
62 uint32_t i, tapCnt, blkCnt; /* Loop counters */
63
64 /* S->pState points to state array which contains previous frame (numTaps - 1) samples */
65 /* pStateCurnt points to the location where the new input data should be written */
66 pStateCurnt = &(S->pState[(numTaps - 1U)]);
67
68 /* Initialize blkCnt with number of taps */
69 blkCnt = blockSize;
70
71 while (blkCnt > 0U)
72 {
73 /* Copy one sample at a time into state buffer */
74 *pStateCurnt++ = *pSrc++;
75
76 /* Set the accumulator to zero */
77 acc0 = 0.;
78
79 /* Initialize state pointer */
80 px = pState;
81
82 /* Initialize Coefficient pointer */
83 pb = pCoeffs;
84
85 i = numTaps;
86
87 /* Perform the multiply-accumulates */
88 while (i > 0U)
89 {
90 /* acc = b[numTaps-1] * x[n-numTaps-1] + b[numTaps-2] * x[n-numTaps-2] + b[numTaps-3] * x[n-numTaps-3] +...+ b[0] * x[0] */
91 acc0 += *px++ * *pb++;
92
93 i--;
94 }
95
96 /* Store result in destination buffer. */
97 *pDst++ = acc0;
98
99 /* Advance state pointer by 1 for the next sample */
100 pState = pState + 1U;
101
102 /* Decrement loop counter */
103 blkCnt--;
104 }
105
106 /* Processing is complete.
107 Now copy the last numTaps - 1 samples to the start of the state buffer.
108 This prepares the state buffer for the next function call. */
109
110 /* Points to the start of the state buffer */
111 pStateCurnt = S->pState;
112
113 /* Initialize tapCnt with number of taps */
114 tapCnt = (numTaps - 1U);
115
116 /* Copy remaining data */
117 while (tapCnt > 0U)
118 {
119 *pStateCurnt++ = *pState++;
120
121 /* Decrement loop counter */
122 tapCnt--;
123 }
124
125}
