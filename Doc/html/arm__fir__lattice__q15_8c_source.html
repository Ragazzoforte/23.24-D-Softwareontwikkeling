92 /* Read g0(n-1) from state buffer */

95 /* Process first sample for first tap */
96 /* f1(n) = f0(n) + K1 * g0(n-1) */
97 fnext0 = (q31_t) ((gcurr0 * (*pk)) >> 15U) + fcurr0;
98 fnext0 = __SSAT(fnext0, 16);

100 /* g1(n) = f0(n) * K1 + g0(n-1) */
101 gnext0 = (q31_t) ((fcurr0 * (*pk)) >> 15U) + gcurr0;
102 gnext0 = __SSAT(gnext0, 16);
103
104 /* Process second sample for first tap */
105 fnext1 = (q31_t) ((fcurr0 * (*pk)) >> 15U) + fcurr1;
106 fnext1 = __SSAT(fnext1, 16);
107 gnext1 = (q31_t) ((fcurr1 * (*pk)) >> 15U) + fcurr0;
108 gnext1 = __SSAT(gnext1, 16);
109
110 /* Read next two samples from input buffer */
111 /* f0(n+2) = x(n+2) */
112 fcurr2 = *pSrc++;
113 fcurr3 = *pSrc++;
114
115 /* Process third sample for first tap */
116 fnext2 = (q31_t) ((fcurr1 * (*pk)) >> 15U) + fcurr2;
117 fnext2 = __SSAT(fnext2, 16);
118 gnext2 = (q31_t) ((fcurr2 * (*pk)) >> 15U) + fcurr1;
119 gnext2 = __SSAT(gnext2, 16);
120
121 /* Process fourth sample for first tap */
122 fnext3 = (q31_t) ((fcurr2 * (*pk )) >> 15U) + fcurr3;
123 fnext3 = __SSAT(fnext3, 16);
124 gnext3 = (q31_t) ((fcurr3 * (*pk++)) >> 15U) + fcurr2;
125 gnext3 = __SSAT(gnext3, 16);
126
127 /* Copy only last input sample into the state buffer
128 which will be used for next samples processing */
129 *px++ = (q15_t) fcurr3;
130
131 /* Update of f values for next coefficient set processing */
132 fcurr0 = fnext0;
133 fcurr1 = fnext1;
134 fcurr2 = fnext2;
135 fcurr3 = fnext3;

138 stageCnt = (numStages - 1U) >> 2U;
139
140 /* Loop over the number of taps. Unroll by a factor of 4.
141 Repeat until we've computed numStages-3 coefficients. */

143 /* Process 2nd, 3rd, 4th and 5th taps ... here */

145 {

148
149 /* save g1(n) in state buffer */
150 *px++ = (q15_t) gnext3;
151
152 /* Process first sample for 2nd, 6th .. tap */
153 /* Sample processing for K2, K6.... */
154 /* f1(n) = f0(n) + K1 * g0(n-1) */
155 fnext0 = (q31_t) ((gcurr0 * (*pk)) >> 15U) + fcurr0;
156 fnext0 = __SSAT(fnext0, 16);
157

159 /* for sample 2 processing */

161 fnext1 = __SSAT(fnext1, 16);
162

165 fnext2 = __SSAT(fnext2, 16);
166

169 fnext3 = __SSAT(fnext3, 16);

208 fcurr3 = __SSAT(fcurr3, 16);
