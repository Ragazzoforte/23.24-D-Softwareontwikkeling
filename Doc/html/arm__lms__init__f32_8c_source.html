55void arm_lms_init_f32(
56 arm_lms_instance_f32 * S,
57 uint16_t numTaps,
58 float32_t * pCoeffs,
59 float32_t * pState,
60 float32_t mu,
61 uint32_t blockSize)
62{
63 /* Assign filter taps */
64 S->numTaps = numTaps;
65
66 /* Assign coefficient pointer */
67 S->pCoeffs = pCoeffs;
68
69 /* Clear state buffer and size is always blockSize + numTaps */
70 memset(pState, 0, (numTaps + (blockSize - 1)) * sizeof(float32_t));
71
72 /* Assign state pointer */
73 S->pState = pState;
74
75 /* Assign Step size value */
76 S->mu = mu;
77}
