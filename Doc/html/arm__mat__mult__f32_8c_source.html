68__STATIC_INLINE arm_status arm_mat_mult_f32_2x2_mve(
69 const arm_matrix_instance_f32 *pSrcA,
70 const arm_matrix_instance_f32 *pSrcB,
71 arm_matrix_instance_f32 *pDst)
72{
73 /* {a00, a00, a10, a10} */
74 static const uint32_t offsetA0[4] = { 0, 0, 2, 2 };
75 /* {b00, b01, b00, b01} */
76 static const uint32_t offsetB0[4] = { 0, 1, 0, 1 };
77 /* {a01, a01, a11, a11} */
78 static const uint32_t offsetA1[4] = { 1, 1, 3, 3 };
79 /* {b10, b11, b10, b11} */
80 static const uint32_t offsetB1[4] = { 2, 3, 2, 3 };
81
82 uint32x4_t vecOffsA, vecOffsB;
83 f32x4_t vecInA, vecInB, vecDst;
84
85 vecOffsA = vldrwq_u32((uint32_t const *) offsetA0);
86 vecOffsB = vldrwq_u32((uint32_t const *) offsetB0);
87
88 vecInA = vldrwq_gather_shifted_offset((float32_t const *) pSrcA->pData, vecOffsA);
89 vecInB = vldrwq_gather_shifted_offset((float32_t const *) pSrcB->pData, vecOffsB);
90
91 vecDst = vmulq(vecInA, vecInB);
92
93 vecOffsA = vldrwq_u32((uint32_t const *) offsetA1);
94 vecOffsB = vldrwq_u32((uint32_t const *) offsetB1);
95
96 vecInA = vldrwq_gather_shifted_offset((float32_t const *) pSrcA->pData, vecOffsA);
97 vecInB = vldrwq_gather_shifted_offset((float32_t const *) pSrcB->pData, vecOffsB);
98
99 vecDst = vfmaq(vecDst, vecInA, vecInB);
100
101 vstrwq_f32(pDst->pData, vecDst);
102
103 return (ARM_MATH_SUCCESS);
104
105}

108/*
109 * A = {{a00, a01, a02},
110 * {a10, a11, a12},
111 * {a20, a21, a22}}
112 * B = {{b00, b01, b02},
113 * {b10, b11, b12},
114 * {b20, b21, b22}}
115 *
116 * Dst = {{a00 b00 + a01 b10 + a02 b20, a00 b01 + a01 b11 + a02 b21, a00 b02 + a01 b12 + a02 b22},
117 * {a10 b00 + a11 b10 + a12 b20, a10 b01 + a11 b11 + a12 b21, a10 b02 + a11 b12 + a12 b22},
118 * {a20 b00 + a21 b10 + a22 b20, a20 b01 + a21 b11 + a22 b21, a20 b02 + a21 b12 + a22 b22}}
119 */
120__STATIC_INLINE arm_status arm_mat_mult_f32_3x3_mve(
121 const arm_matrix_instance_f32 *pSrcA,
122 const arm_matrix_instance_f32 *pSrcB,
123 arm_matrix_instance_f32 *pDst)
124{
125 float32_t *pInB = pSrcB->pData; /* input data matrix pointer B */
126 float32_t *pInA = pSrcA->pData; /* input data matrix pointer A */
127 float32_t *pOut = pDst->pData; /* output data matrix pointer */
128 float32_t *pInA0, *pInA1, *pInA2;
129 f32x4_t vecMac0, vecMac1, vecMac2;
130 f32x4_t vecInB;
131 float32_t const *pSrBVec;
132
133 pSrBVec = (float32_t const *) pInB;
134
135 pInA0 = pInA;
136 pInA1 = pInA0 + MATRIX_DIM3;
137 pInA2 = pInA1 + MATRIX_DIM3;
138 /* enable predication to disable last (4th) vector element */
139 mve_pred16_t p0 = vctp32q(MATRIX_DIM3);
140
141 /*
142 * load {b0,0, b0,1, b0,2, 0}
143 */
144 vecInB = vldrwq_z_f32(pSrBVec, p0);
145 pSrBVec += MATRIX_DIM3;
146
147 vecMac0 = vmulq(vecInB, *pInA0++);
148 vecMac1 = vmulq(vecInB, *pInA1++);
149 vecMac2 = vmulq(vecInB, *pInA2++);
150 /*
151 * load {b1,0, b1,1, b1,2, 0}
152 */
153 vecInB = vldrwq_z_f32(pSrBVec, p0);
154 pSrBVec += MATRIX_DIM3;
155
156 vecMac0 = vfmaq(vecMac0, vecInB, *pInA0++);
157 vecMac1 = vfmaq(vecMac1, vecInB, *pInA1++);
158 vecMac2 = vfmaq(vecMac2, vecInB, *pInA2++);
159 /*
160 * load {b2,0, b2,1 , b2,2, 0}
161 */
162 vecInB = vldrwq_z_f32(pSrBVec, p0);
163 pSrBVec += MATRIX_DIM3;
164
165 vecMac0 = vfmaq(vecMac0, vecInB, *pInA0++);
166 vecMac1 = vfmaq(vecMac1, vecInB, *pInA1++);
167 vecMac2 = vfmaq(vecMac2, vecInB, *pInA2++);
168
169 /* partial vector stores */
170 vstrwq_p_f32(pOut, vecMac0, p0);
171 pOut += MATRIX_DIM3;
172 vstrwq_p_f32(pOut, vecMac1, p0);
173 pOut += MATRIX_DIM3;
174 vstrwq_p_f32(pOut, vecMac2, p0);
175 /*
176 * Return to application
177 */
178 return (ARM_MATH_SUCCESS);
179}

184__STATIC_INLINE arm_status arm_mat_mult_f32_4x4_mve(
185 const arm_matrix_instance_f32 *pSrcA,
186 const arm_matrix_instance_f32 *pSrcB,
187 arm_matrix_instance_f32 *pDst)
188{
189 float32_t const *pSrBVec;
190 float32_t *pInB = pSrcB->pData; /* input data matrix pointer B */
191 float32_t *pInA = pSrcA->pData; /* input data matrix pointer A */
192 float32_t *pOut = pDst->pData; /* output data matrix pointer */
193 float32_t *pInA0, *pInA1, *pInA2, *pInA3;
194 f32x4_t vecMac0, vecMac1, vecMac2, vecMac3;
195 f32x4_t vecInB;
196
197 pSrBVec = (float32_t const *) pInB;
198
199 pInA0 = pInA;
200 pInA1 = pInA0 + MATRIX_DIM4;
201 pInA2 = pInA1 + MATRIX_DIM4;
202 pInA3 = pInA2 + MATRIX_DIM4;
203 /*
204 * load {b0,0, b0,1, b0,2, b0,3}
205 */
206 vecInB = vld1q(pSrBVec);
207 pSrBVec += MATRIX_DIM4;
208
209 vecMac0 = vmulq(vecInB, *pInA0++);
210 vecMac1 = vmulq(vecInB, *pInA1++);
211 vecMac2 = vmulq(vecInB, *pInA2++);
212 vecMac3 = vmulq(vecInB, *pInA3++);
213 /*
214 * load {b1,0, b1,1, b1,2, b1,3}
215 */
216 vecInB = vld1q(pSrBVec);
217 pSrBVec += MATRIX_DIM4;
218
219 vecMac0 = vfmaq(vecMac0, vecInB, *pInA0++);
220 vecMac1 = vfmaq(vecMac1, vecInB, *pInA1++);
221 vecMac2 = vfmaq(vecMac2, vecInB, *pInA2++);
222 vecMac3 = vfmaq(vecMac3, vecInB, *pInA3++);
223 /*
224 * load {b2,0, b2,1, b2,2, b2,3}
225 */
226 vecInB = vld1q(pSrBVec);
227 pSrBVec += MATRIX_DIM4;
228
229 vecMac0 = vfmaq(vecMac0, vecInB, *pInA0++);
230 vecMac1 = vfmaq(vecMac1, vecInB, *pInA1++);
231 vecMac2 = vfmaq(vecMac2, vecInB, *pInA2++);
232 vecMac3 = vfmaq(vecMac3, vecInB, *pInA3++);
233 /*
234 * load {b3,0, b3,1, b3,2, b3,3}
235 */
236 vecInB = vld1q(pSrBVec);
237 pSrBVec += MATRIX_DIM4;
238
239 vecMac0 = vfmaq(vecMac0, vecInB, *pInA0++);
240 vecMac1 = vfmaq(vecMac1, vecInB, *pInA1++);
241 vecMac2 = vfmaq(vecMac2, vecInB, *pInA2++);
242 vecMac3 = vfmaq(vecMac3, vecInB, *pInA3++);
243
244 vst1q(pOut, vecMac0);
245 pOut += MATRIX_DIM4;
246 vst1q(pOut, vecMac1);
247 pOut += MATRIX_DIM4;
248 vst1q(pOut, vecMac2);
249 pOut += MATRIX_DIM4;
250 vst1q(pOut, vecMac3);
251 /*
252 * Return to application
253 */
254 return (ARM_MATH_SUCCESS);
255}

266arm_status arm_mat_mult_f32(
267 const arm_matrix_instance_f32 * pSrcA,
268 const arm_matrix_instance_f32 * pSrcB,
269 arm_matrix_instance_f32 * pDst)
270{
271 float32_t *pInB = pSrcB->pData; /* input data matrix pointer B */
272 float32_t *pInA = pSrcA->pData; /* input data matrix pointer A */
273 float32_t *pOut = pDst->pData; /* output data matrix pointer */
274 int numRowsA = pSrcA->numRows; /* number of rows of input matrix A */
275 int numColsB = pSrcB->numCols; /* number of columns of input matrix B */
276 int numColsA = pSrcA->numCols; /* number of columns of input matrix A */
277 uint32_t blkCnt; /* loop counters */
278 uint32_t i;
279 arm_status status;
280
281#ifdef ARM_MATH_MATRIX_CHECK
282
283 /* Check for matrix mismatch condition */
284 if ((pSrcA->numCols != pSrcB->numRows) ||
285 (pSrcA->numRows != pDst->numRows) || (pSrcB->numCols != pDst->numCols))
286 {
287 /* Set status as ARM_MATH_SIZE_MISMATCH */
288 status = ARM_MATH_SIZE_MISMATCH;
289 }
290 else
291#endif /* #ifdef ARM_MATH_MATRIX_CHECK */
292 {
293 /* small squared matrix specialized routines */
294 if(numRowsA == numColsB && numColsB == numColsA) {
295 if (numRowsA == 1)
296 {
297 pOut[0] = pInA[0] * pInB[0];
298 return(ARM_MATH_SUCCESS);
299 }
300 else if(numRowsA == 2)
301 return arm_mat_mult_f32_2x2_mve(pSrcA, pSrcB, pDst);
302 else if(numRowsA == 3)
303 return arm_mat_mult_f32_3x3_mve(pSrcA, pSrcB, pDst);
304 else if(numRowsA == 4)
305 return arm_mat_mult_f32_4x4_mve(pSrcA, pSrcB, pDst);
306 }
307
308 /* main loop process 4 rows */
309 i = numRowsA >> 2;
310 while (i > 0U)
311 {
312 float32_t *pInA0, *pInA1, *pInA2, *pInA3;
313 float32_t *pInB0;
314 float32_t *pOut0, *pOut1, *pOut2, *pOut3;
315 f32x4_t vecMac0, vecMac1, vecMac2, vecMac3;
316 f32x4_t vecInB;
317
318 /* pointers to 4 consecutive output rows */
319 pOut0 = pOut;
320 pOut1 = pOut0 + numColsB;
321 pOut2 = pOut1 + numColsB;
322 pOut3 = pOut2 + numColsB;
323 pInB0 = pInB;
324
325 uint32_t k = numColsB >> 2;
326 while (k > 0U)
327 {
328 /* pointers to 4 consecutive Matrix A rows */
329 pInA0 = pInA;
330 pInA1 = pInA0 + numColsA;
331 pInA2 = pInA1 + numColsA;
332 pInA3 = pInA2 + numColsA;
333
334 vecMac0 = vdupq_n_f32(0.0f);
335 vecMac1 = vdupq_n_f32(0.0f);
336 vecMac2 = vdupq_n_f32(0.0f);
337 vecMac3 = vdupq_n_f32(0.0f);
338
339 blkCnt = numColsA;
340
341 while (blkCnt > 0U)
342 {
343 /*
344 * load {bi,4n+0, bi,4n+1, bi,4n+2, bi,4n+3}
345 */
346 vecInB = *(f32x4_t *)pInB0; /* vldrwq_f32(pInB0, 0); */
347
348 vecMac0 = vfmaq(vecMac0, vecInB, *pInA0++);
349 vecMac1 = vfmaq(vecMac1, vecInB, *pInA1++);
350 vecMac2 = vfmaq(vecMac2, vecInB, *pInA2++);
351 vecMac3 = vfmaq(vecMac3, vecInB, *pInA3++);
352
353 pInB0 = pInB0 + numColsB;
354 /*
355 * Decrement the blockSize loop counter
356 */
357 blkCnt--;
358 }
359
360 /* Store the results (4 x 4 block) in the destination buffer */
361 vst1q(pOut0, vecMac0);
362 pOut0 += 4;
363 vst1q(pOut1, vecMac1);
364 pOut1 += 4;
365 vst1q(pOut2, vecMac2);
366 pOut2 += 4;
367 vst1q(pOut3, vecMac3);
368 pOut3 += 4;
369
370 /*
371 * rewind
372 */
373 pInB0 -= (numColsB * numColsA) - 4;
374 k--;
375 }
376
377 int colBLeft = numColsB & 3;
378 if (colBLeft)
379 {
380 pInA0 = pInA;
381 pInA1 = pInA0 + numColsA;
382 pInA2 = pInA1 + numColsA;
383 pInA3 = pInA2 + numColsA;
384 mve_pred16_t p0 = vctp32q(colBLeft);
385
386 vecMac0 = vdupq_n_f32(0.0f);
387 vecMac1 = vdupq_n_f32(0.0f);
388 vecMac2 = vdupq_n_f32(0.0f);
389 vecMac3 = vdupq_n_f32(0.0f);
390
391 blkCnt = numColsA;
392
393 while (blkCnt > 0U)
394 {
395 /*
396 * load {bi,4n+0, bi,4n+1, bi,4n+2, bi,4n+3}
397 */
398 vecInB = vldrwq_z_f32(pInB0, p0);
399
400 vecMac0 = vfmaq(vecMac0, vecInB, *pInA0++);
401 vecMac1 = vfmaq(vecMac1, vecInB, *pInA1++);
402 vecMac2 = vfmaq(vecMac2, vecInB, *pInA2++);
403 vecMac3 = vfmaq(vecMac3, vecInB, *pInA3++);
404
405 pInB0 = pInB0 + numColsB;
406 /*
407 * Decrement the blockSize loop counter
408 */
409 blkCnt--;
410 }
411
412 /* Store the results (4 x colBLeft block) in the destination buffer */
413 vstrwq_p_f32(pOut0, vecMac0, p0);
414 vstrwq_p_f32(pOut1, vecMac1, p0);
415 vstrwq_p_f32(pOut2, vecMac2, p0);
416 vstrwq_p_f32(pOut3, vecMac3, p0);
417 }
418
419 /* move to next rows */
420 pInA += 4 * numColsA;
421 pOut += 4 * numColsB;
422 i--;
423 }
424
425 /*
426 * non multiple of 4 rows for Matrix A
427 * process single row
428 */
429 if (numRowsA & 3)
430 {
431 i = numRowsA & 3;
432 while (i > 0U)
433 {
434 float32_t *pInA0;
435 float32_t *pInB0;
436 float32_t *pOut0;
437 f32x4_t vecInB;
438 f32x4_t vecMac0;
439
440 pOut0 = pOut;
441 pInB0 = pInB;
442
443 uint32_t k = numColsB >> 2;
444 while (k > 0U)
445 {
446 pInA0 = pInA;
447
448 vecMac0 = vdupq_n_f32(0.0f);
449 blkCnt = numColsA;
450 while (blkCnt > 0U)
451 {
452 /*
453 * load {bi,4n+0, bi,4n+1, bi,4n+2, bi,4n+3}
454 */
455 vecInB = *(f32x4_t *)pInB0; /* vldrwq_f32(pInB0, 0); */
456
457 vecMac0 = vfmaq(vecMac0, vecInB, *pInA0++);
458
459 pInB0 = pInB0 + numColsB;
460 /*
461 * Decrement the blockSize loop counter
462 */
463 blkCnt--;
464 }
465
466 /* Store the results (1 x 4 block) in the destination buffer */
467 vst1q(pOut0, vecMac0);
468 pOut0 += 4;
469
470 /*
471 * rewind
472 */
473 pInB0 -= (numColsB * numColsA) - 4;
474 k--;
475 }
476
477 int colBLeft = numColsB & 3;
478 if (colBLeft)
479 {
480 pInA0 = pInA;
481 mve_pred16_t p0 = vctp32q(colBLeft);
482
483 vecMac0 = vdupq_n_f32(0.0f);
484 blkCnt = numColsA;
485 while (blkCnt > 0U)
486 {
487 /*
488 * load {bi,4n+0, bi,4n+1, bi,4n+2, bi,4n+3}
489 */
490 vecInB = vldrwq_z_f32(pInB0, p0);
491
492 vecMac0 = vfmaq(vecMac0, vecInB, *pInA0++);
493
494 pInB0 = pInB0 + numColsB;
495 /*
496 * Decrement the blockSize loop counter
497 */
498 blkCnt--;
499 }
500 /* Store the results (1 x colBLeft block) in the destination buffer */
501 vstrwq_p_f32(pOut0, vecMac0, p0);
502 }
503
504 /* move to next row */
505 pInA += 1 * numColsA;
506 pOut += 1 * numColsB;
507 i--;
508 }
509
510 }
511 status = ARM_MATH_SUCCESS;
512 }
513
514 /* Return to application */
515 return (status);
516}

528arm_status arm_mat_mult_f32(
529 const arm_matrix_instance_f32 * pSrcA,
530 const arm_matrix_instance_f32 * pSrcB,
531 arm_matrix_instance_f32 * pDst)
532{
533 float32_t *pIn1 = pSrcA->pData; /* input data matrix pointer A */
534 float32_t *pIn2 = pSrcB->pData; /* input data matrix pointer B */
535 float32_t *pInA = pSrcA->pData; /* input data matrix pointer A */
536 float32_t *pOut = pDst->pData; /* output data matrix pointer */
537 float32_t *px; /* Temporary output data matrix pointer */
538 float32_t sum; /* Accumulator */
539 uint16_t numRowsA = pSrcA->numRows; /* number of rows of input matrix A */
540 uint16_t numColsB = pSrcB->numCols; /* number of columns of input matrix B */
541 uint16_t numColsA = pSrcA->numCols; /* number of columns of input matrix A */
542
543
544 uint16_t col, i = 0U, j, row = numRowsA, rowCnt, colCnt; /* loop counters */
545 arm_status status; /* status of matrix multiplication */
546
547 float32x4_t a0V, a1V, a2V, a3V, a4V, a5V, a6V, a7V;
548 float32x4_t acc0,acc1,acc2,acc3,acc4,acc5,acc6,acc7,temp;
549 float32x2_t accum = vdup_n_f32(0);
550 float32_t *pIn1B = pSrcA->pData;
551 float32_t *pIn1C = pSrcA->pData;
552 float32_t *pIn1D = pSrcA->pData;
553 float32_t *pIn1E = pSrcA->pData;
554 float32_t *pIn1F = pSrcA->pData;
555 float32_t *pIn1G = pSrcA->pData;
556 float32_t *pIn1H = pSrcA->pData;
557
558 float32_t *pxB,*pxC, *pxD, *pxE, *pxF, *pxG, *pxH; /* Temporary output data matrix pointer */
559 float32_t sum0,sum1, sum2,sum3, sum4, sum5 , sum6, sum7;
560
561#ifdef ARM_MATH_MATRIX_CHECK
562
563 /* Check for matrix mismatch condition */
564 if ((pSrcA->numCols != pSrcB->numRows) ||
565 (pSrcA->numRows != pDst->numRows) || (pSrcB->numCols != pDst->numCols))
566 {
567 /* Set status as ARM_MATH_SIZE_MISMATCH */
568 status = ARM_MATH_SIZE_MISMATCH;
569 }
570 else
571#endif /* #ifdef ARM_MATH_MATRIX_CHECK */
572 {
573 /* The following loop performs the dot-product of each row in pSrcA with each column in pSrcB */
574 /* Row loop */
575 rowCnt = row >> 3;
576
577 while(rowCnt > 0)
578 {
579 /* Output pointer is set to starting address of the row being processed */
580 px = pOut + GROUPOFROWS*i;
581 pxB = px + numColsB;
582 pxC = px + 2*numColsB;
583 pxD = px + 3*numColsB;
584 pxE = px + 4*numColsB;
585 pxF = px + 5*numColsB;
586 pxG = px + 6*numColsB;
587 pxH = px + 7*numColsB;
588
589 /* For every row wise process, the column loop counter is to be initiated */
590 col = numColsB;
591
592 /* For every row wise process, the pIn2 pointer is set
593 ** to the starting address of the pSrcB data */
594 pIn2 = pSrcB->pData;
595
596 j = 0U;
597
598 /* Column loop */
599 do
600 {
601 /* Set the variable sum, that acts as accumulator, to zero */
602 sum0 = 0.0f;
603 sum1 = 0.0f;
604 sum2 = 0.0f;
605 sum3 = 0.0f;
606 sum4 = 0.0f;
607 sum5 = 0.0f;
608 sum6 = 0.0f;
609 sum7 = 0.0f;
610
611 /* Initiate the pointer pIn1 to point to the starting address of the column being processed */
612 pIn1 = pInA;
613 pIn1B = pIn1 + numColsA;
614 pIn1C = pIn1 + 2*numColsA;
615 pIn1D = pIn1 + 3*numColsA;
616 pIn1E = pIn1 + 4*numColsA;
617 pIn1F = pIn1 + 5*numColsA;
618 pIn1G = pIn1 + 6*numColsA;
619 pIn1H = pIn1 + 7*numColsA;
620
621 acc0 = vdupq_n_f32(0.0);
622 acc1 = vdupq_n_f32(0.0);
623 acc2 = vdupq_n_f32(0.0);
624 acc3 = vdupq_n_f32(0.0);
625 acc4 = vdupq_n_f32(0.0);
626 acc5 = vdupq_n_f32(0.0);
